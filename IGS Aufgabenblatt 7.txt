task main(){
  while(true){
	if(ButtonPressed(1,false)){
		OnFwd(OUT_A,25);
	}
	if(ButtonPressed(2,false)){
		OnRev(OUT_A,25);
	}
	if(ButtonPressed(3,false)){
		break;
	}
	Wait(5);
	Off(Out_A);
  }
}
